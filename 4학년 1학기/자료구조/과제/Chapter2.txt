5. Complex add_complex(Complex a, Complex b)
{
	Complex result;

	result.real = a.real + b.real;
	result.imag = a.imag + b.imag;

	return result;
}

6. void insert(int array[], int loc, int value) {
	int i = 0;

	for (i = ITEMS - 2; i >= loc; i--)
	{
		array[i + 1] = array[i];
	}

	array[loc] = value;
}
